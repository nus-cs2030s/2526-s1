Maybe<Internship> match(Resume r) {
  if (r == null) {
    return Maybe.none();
  }
  Maybe<List<String>> optList
      = r.getListOfLanguages();
  List<String> list;
  if (optList.equals(Maybe.none())) {
    list = List.of();
  } else {
    list = optList.get(); // cannot call
  }
  if (list.contains("Java")) {
    return Maybe.of(findInternship(list));
  } else {
    return Maybe.none();
  }
}
